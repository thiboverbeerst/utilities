CREATE DATABASE beersdb;

USE beersdb;
	
CREATE TABLE brewers(
	brewerno INT AUTO_INCREMENT NOT NULL,
	brname NVARCHAR(50) NULL,
	address NVARCHAR(50) NULL,
	zip SMALLINT NULL,
	city NVARCHAR(50) NULL,
	turnover INT NULL,
	CONSTRAINT PK_brewers PRIMARY KEY(brewerno)
)ENGINE = INNODB;

CREATE TABLE beertypes(
	typeno INT AUTO_INCREMENT NOT NULL,
	beertype NVARCHAR(50) NULL,
	CONSTRAINT PK_beertypes PRIMARY KEY(typeno)  
)ENGINE = INNODB;

CREATE TABLE beers(
	beerno INT AUTO_INCREMENT NOT NULL,
	bname NVARCHAR(100) NULL,
	brewerno INT NULL,
	typeno INT NULL,
	alcohol REAL NULL,
	CONSTRAINT PK_beers PRIMARY KEY(beerno), 
	CONSTRAINT FK1_beers FOREIGN KEY(brewerno) REFERENCES brewers(brewerno),
	CONSTRAINT FK2_beers FOREIGN KEY(typeno) REFERENCES beertypes(typeno)
)ENGINE = INNODB;

CREATE TABLE persons(
	personno INT AUTO_INCREMENT NOT NULL,
	pname NVARCHAR(100) NULL,
	gender NVARCHAR(1) NULL,
	birthdate DATE NULL,
	deathdate DATE NULL,
	CONSTRAINT PK_persons PRIMARY KEY(personno)  
)ENGINE = INNODB;

CREATE TABLE beertasters(
	personno INT NOT NULL,
	beerno INT NOT NULL,
	remarks NVARCHAR(1000) NULL,
	CONSTRAINT PK_beertasters PRIMARY KEY(personno, beerno), 
	CONSTRAINT FK1_beertasters FOREIGN KEY(personno) REFERENCES persons(personno),
	CONSTRAINT FK2_beertasters FOREIGN KEY(beerno) REFERENCES beers(beerno)
)ENGINE = INNODB;
